これをまとめてリスト化 charlist = ascii_chars + halfwidth_kana + funcspclist print(charlist) def singlebtn(btnno): btns = [c0, c1, c2, c3, c4, c5, c6, c7, c8, c9] obj = btns[btnno] rtrn = obj.value() return rtrn def readbtns(): for i in range(10): pu = singlebtn(i) if pu == 0: return i return None def waitbtn(): while True: rb = readbtns() if rb is not None: return rb def charin(takeit): lcd.clear() while True: print(c1.value(), c2.value(), c3.value(), c4.value(), c5.value(), c6.value(), c7.value(), c8.value(), c9.value(), c0.value()) sleep(0.3) btnin1 = waitbtn() sleep(0.3) btnin2 = waitbtn() sleep(0.3) btnin3 = waitbtn() btnin1 = str(btnin1) btnin2 = str(btnin2) btnin3 = str(btnin3) btnina = btnin1 + btnin2 + btnin3 btnina = int(btnina) btnina = btnina % 159 print(btnina) rtrn = charlist[btnina] print(rtrn) if rtrn == "EN": return takeit if rtrn == "BS" and len(takeit) is not 0: takeit = takeit[:-1] else: takeit = takeit + rtrn lcd.move_to(0, 0) lcd.clear() lcd.putstr(takeit) a = "" charin(a)
